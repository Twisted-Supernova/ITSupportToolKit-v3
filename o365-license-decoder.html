PowerShell Commands for License Management

# Connect to Microsoft Graph Connect-MgGraph -Scopes "Organization.Read.All" # Get all available licenses in your tenant Get-MgSubscribedSku | Select SkuPartNumber, SkuId, ConsumedUnits -ExpandProperty PrepaidUnits # Get user's assigned licenses Get-MgUserLicenseDetail -UserId user@domain.com | Select SkuPartNumber, SkuId # Export all users with their licenses Get-MgUser -All -Property DisplayName,UserPrincipalName,AssignedLicenses | Select DisplayName, UserPrincipalName, @{N="Licenses";E={($_.AssignedLicenses.SkuId -join '; ')}}
